49 #define RECOVER_FIELD(FNAME,TYPE,FROM,TO,DEF) \
50  TYPE get_##FNAME(const std::string& line) const { \
51  if (line.length() < TO){\
52  if (! Mandatory_fields::FNAME) return TYPE(DEF);\
53  else{\
54  std::cerr << "Fatal error: Cannot extract field \'" << #FNAME << "\' in \n";\
55  std::cerr << "<|" << line << "|>\n";\
56  exit (EXIT_FAILURE);\
57  }\
58  }\
59  return PDB::extract_field<TYPE>(line,FROM,TO,DEF,#FNAME,Mandatory_fields::FNAME); \
60  }

69  template<class T>
70  T extract_field(const std::string& line,unsigned from, unsigned to,T default_value,const char* name,bool is_mandatory=false){
71  std::string s=line.substr(from,to-from+1);
72  boost::trim(s);
73  if (s.length()==0){
74  if (!is_mandatory)
75  return default_value;
76  else{
77  std::cerr << "Fatal error: field \'"<< name << "\' is empty in \n";
78  std::cerr << "<|" << line << "|>\n";
79  exit (EXIT_FAILURE);
80  }
81  }
82  try{
83  return boost::lexical_cast<T>(s);
84  }
85  catch(boost::bad_lexical_cast &)
86  {
87  std::cerr << "Fatal error: Cannot convert \'"<< s << "\' to " << typeid(T).name() << " in line\n";
88  std::cerr << "<|" << line << "|>\n";
89  exit (EXIT_FAILURE);
90  }
91  }

164  int get_charge(const std::string& line) const {
165  std::string charge_str=get_charge_str(line);
166  if (charge_str==" ")
167  return NO_CHARGE;
168  else{
169  int ch=1;
170  if (charge_str.size()==2){
171  if (charge_str[1]=='-') ch*=-1;
172  else{
173  if (charge_str[1]!='+'){
174  std::cerr << "Fatal error: Cannot extract field \'charge\' in \n";
175  std::cerr << "<|" << line << "|>\n";\
176  exit(EXIT_FAILURE);
177  }
178  }
179  }
180  try{
181  return ch*boost::lexical_cast<int>(charge_str[0]);
182  }
183  catch(boost::bad_lexical_cast &){
184  std::cerr << "Fatal error: Cannot convert \'"<< charge_str[0] << "\' to " << typeid(int).name() << " in line\n";
185  std::cerr << "<|" << line << "|>\n";
186  exit (EXIT_FAILURE);
187  }
188  }
189  }

205  template<class PDB_Atom>
206  std::string
207  get_atom_pdb_format(const PDB_Atom& atom)
208  {
209  std::stringstream buf(std::stringstream::out);
210  buf << ( !atom.is_hetatm() ? "ATOM ":"HETATM" );
211  buf << boost::format("%5d %4s%c%3s %c%4d%c %8.3f%8.3f%8.3f%6.2f%6.2f %2s")
212  % atom.atom_serial_number()
213  % atom.atom_name().c_str()
214  % atom.alternate_location()
215  % atom.residue_name().c_str()
216  % atom.chain_identifier()
217  % atom.residue_sequence_number()
218  % atom.insertion_code()
219  % atom.x()
220  % atom.y()
221  % atom.z()
222  % atom.occupancy()
223  % atom.temperature_factor()
224  % atom.element().c_str();
225  if (atom.charge()==NO_CHARGE) buf << " ";
226  else buf << boost::format("% 2i") % atom.charge();
227  return buf.str();
228  }
