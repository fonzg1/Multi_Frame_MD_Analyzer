379  Generic_classifier(const std::string& filename){
380  enum Read_state {PROPERTIES, CLASSIFICATION, END, EXTRA,DEFAULT};
381 
382  std::ifstream input(filename.c_str());
383 
384  if (!input){
385  std::cerr << "Cannot open file " << filename << std::endl;
386  exit( EXIT_FAILURE);
387  }
388 
389  std::string buffer;
390  Read_state state=END;
391 
392  std::map<unsigned,Properties> properties_map;
393  unsigned max_property=0;
394 
395 
396  while(input){
397  std::getline(input,buffer); //read a complete line
398 
399  if (!input) {
400  if (input.eof()) break;
401  std::cerr << "Fatal error: Error while reading file "<< filename <<"\n";
402  exit( EXIT_FAILURE );
403  }
404 
405  boost::trim(buffer); //remove white space at the beginning and at the end of the line
406  if (buffer.empty() || buffer[0] == '#') continue; //handle comments and empty lines
407 
408  switch (state){
409  case END:
410  if (buffer == "PROPERTIES") {state=PROPERTIES; break;}
411  if (buffer == "CLASSIFICATION") {state=CLASSIFICATION; break;}
412  if (buffer == "EXTRA") {state=EXTRA; break;}
413  if (buffer == "DEFAULT") {state=DEFAULT; break;}
414  std::cerr << "Fatal error: Error while reading file "<< filename;
415  std::cerr << ", unexpected <|" << buffer << "|> found.\n";
416  exit( EXIT_FAILURE );
417  case CLASSIFICATION:
418  {
419  if (buffer=="END") {state=END; break;}
420  std::stringstream ss (std::stringstream::in | std::stringstream::out);
421  ss << buffer;
422  //inside a function
423  unsigned index=Properties::add_classification(ss,hmap_);
424  if (max_property < index) max_property=index;
425  break;
426  }
427  case PROPERTIES:
428  {
429  if (buffer=="END") {state=END; break;}
430  std::stringstream ss (std::stringstream::in | std::stringstream::out);
431  ss << buffer;
432  unsigned index;
433  ss >> index;
434  //inside a function
435  assert(properties_map.find(index)==properties_map.end());
436  properties_map.insert(std::make_pair(index,Properties(ss,index)));
437  break;
438  }
439  case DEFAULT:
440  {
441  if (buffer=="END") {state=END; break;}
442  unsigned i;
443  std::stringstream ss (std::stringstream::in | std::stringstream::out);
444  ss << buffer;
445  ss >> i;
446  if ( max_property<i ) max_property=i;
447  Properties::index_of_default()=i;
448  break;
449  }
450  case EXTRA:
451  {
452  if (buffer=="END") {state=END; break;}
453  //call some function
454  std::stringstream ss (std::stringstream::in | std::stringstream::out);
455  ss << buffer;
456  Properties::handle_extra(ss);
457  break;
458  }
459  };
460  }
461 
462 
463  unsigned i=0;
464  for (typename std::map<unsigned,Properties>::const_iterator it=properties_map.begin();it!=properties_map.end();++it){
465  if (i++ != it->first){
466  std::cerr << "Fatal error: properties number " << i-1 << " could not be found in file " << filename <<"\n";
467  exit( EXIT_FAILURE );
468  }
469  properties_.push_back(it->second);
470  }
471 
472  if (i <= max_property){
473  std::cerr << "Fatal error: Up to " << max_property+1 << " properties are used, but only " << i << " are declared.\n";
474  exit( EXIT_FAILURE );
475  }
476 
477  max_index_=i;
478 
479  }

485  assert(i <= max_index_);

487  }

493  typename Internal_map_type::const_iterator it=hmap_.find( Properties_::make_key(query) );
494  if ( it==hmap_.end() ){

498  std::cerr << " and no default have been defined.\n";
499  exit( EXIT_FAILURE );
500  }
501  return properties_[it->second];
