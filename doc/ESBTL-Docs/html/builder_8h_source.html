107  if (system_info==RMK){
108  if (line_format.record_type()==PDB::MODEL)
109  current_model=line_format.get_model_number(line);
110  }
111  else{
112  //ensure that references remains valid. You must use the function reserve for this purpose.

114  if (!prebuilt_systems && systems_.size()<(unsigned) system_info)
115  systems_.push_back(System(system_info));
116  systems_[system_info-1].interpret_line(line_format,line,current_model);
117  }
