102 template<class Atom>
103 struct Color_of_atom{
107  std::string operator() (const Atom& atm) const{
108  std::string name=atm.residue().residue_name();
109  if ( name == "ALA" || name == "CYS" || name == "GLY" ||
110  name == "PRO" || name == "SER" || name == "THR" )
111  return std::string("1,1,0"); // yellow: normal
112 
113  if ( name == "VAL" || name == "LEU" || name == "ILE" ||
114  name == "MET" || name == "MSE" || name == "PHE" ||
115  name == "TYR" || name == "TRP" )
116  return std::string("0,0.80,0");// green: hydrophobic
117 
118  if ( name == "HIS" || name == "LYS" || name == "ARG")
119  return std::string("0,0,0.93");//blue: positive charge
120 
121  if ( name == "ASN" || name == "GLN" )
122  return std::string("0.54,0.04,0.31 ");//purple: +/- charge
123 
124  if ( name == "GLU" || name == "ASP" )
125  return std::string("0.77,0,0 ");//red: negative charge
126  return std::string("0.5,0.5,0.5 ");//unknown residue
127  }
128 };
