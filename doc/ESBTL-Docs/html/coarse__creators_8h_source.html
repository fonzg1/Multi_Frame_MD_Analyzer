69 template <class Residue>
70 class Coarse_creator_two_barycenters{
71  typedef typename Residue::Coarse_atom Coarse_atom;
72  typedef typename Residue::Atom Atom;

89  //returns the number of atoms created
90  template <class Output_iterator>
91  int operator()(const Residue& res,Output_iterator out) const{
92  Coarse_atom bb(0,res);
93  Coarse_atom sc(1,res);
94  FT Xbb[4]={0,0,0,0};
95  FT Xsc[4]={0,0,0,0};
96  unsigned tot_res=0;
97  for(typename Residue::Atoms_const_iterator
98  at_it=res.atoms_begin();
99  at_it!=res.atoms_end();
100  ++at_it)
101  {
102  ++tot_res;
103  if ( is_backbone(*at_it) ){
104  bb.add(*at_it);
105  update_coordinates(Xbb,*at_it);
106  }
107  else{
108  sc.add(*at_it);
109  update_coordinates(Xsc,*at_it);
110  }
111  }
112  if (tot_res==0)
113  return 0;
114  unsigned ret=0;
115  if (Xbb[3]!=0){ //to handle case of a ligand for example.
116  static_cast<Point_3&>(bb)=Point_3(Xbb[0]/Xbb[3],Xbb[1]/Xbb[3],Xbb[2]/Xbb[3]);
117  *out++=bb;
118  ++ret;
119  }
120  if (Xsc[3]!=0){
121  static_cast<Point_3&>(sc)=Point_3(Xsc[0]/Xsc[3],Xsc[1]/Xsc[3],Xsc[2]/Xsc[3]);
122  *out++=sc;
123  return ret+1;
124  }
125  else{assert(res.residue_name()=="GLY");}
126  assert(ret==1);
127  return 1;
128  }
129 };

144 class Coarse_creator_closest_to_barycenter{
145  typedef typename Residue::Coarse_atom Coarse_atom;

147  typedef FT_ FT;

151  void update_coordinates(FT* X,const Atom& atom) const{
152  X[0]+=atom.x();
153  X[1]+=atom.y();
154  X[2]+=atom.z();
155  X[3]+=1;

158  static inline FT square(const FT& n){return n*n;}

160  template <class Point_3>
161  static inline FT squared_distance(const Point_3& p1,const Point_3& p2){
162  return square(p1.x()-p2.x())+square(p1.y()-p2.y())+square(p1.z()-p2.z());
163  }
164 
165 public:
166 
167  static const std::string info(){return info_;}
168 
169  //returns the number of atoms created
170  template <class Output_iterator>
171  int operator()(const Residue& res,Output_iterator out) const {
172  Coarse_atom c_atm(0,res);
173  FT X_c_atm[4]={0,0,0,0};
174  std::list<const Atom*> candidate_list;
175  for(typename Residue::Atoms_const_iterator
176  at_it=res.atoms_begin();
177  at_it!=res.atoms_end();
178  ++at_it)
179  {
180  if ( is_side_chain_or_CA(*at_it) ){
181  c_atm.add(*at_it);
182  update_coordinates(X_c_atm,*at_it);
183  candidate_list.push_back( &(*at_it) );
184  }
185  }
186  assert(X_c_atm[3]!=0);
187  Atom barycenter=Atom(X_c_atm[0]/X_c_atm[3],X_c_atm[1]/X_c_atm[3],X_c_atm[2]/X_c_atm[3]);
188  FT min_dist=std::numeric_limits<FT>::max();
189 
190  typename std::list<const Atom*>::iterator min_it=candidate_list.end();
191  for (typename std::list<const Atom*>::iterator it=candidate_list.begin();it!=candidate_list.end();++it){
192  FT curr_dist=squared_distance(barycenter,*(*it));
193  if (curr_dist < min_dist){
194  min_it=it;
195  min_dist=curr_dist;
196  }
197  }
198  assert(min_it!=candidate_list.end());
199  static_cast<typename Coarse_atom::Point_3&>(c_atm)=static_cast<const typename Atom::Point_3&>(*(*min_it));
200 
201 
202  *out++=c_atm;
203  return 1;
204  }
205 };
