168 template<class Input_iterator,class System>
169 void insert_coarse_atoms(Input_iterator first,Input_iterator last,System& system,int modelid=1,char chainid='Z',std::string resname="SOL",int starting_res_index=1){
170  typename System::Model& model=system.get_or_create_model(modelid);
171  typename System::Chain& chain=model.get_or_create_chain(chainid);
172  int res_index=starting_res_index-1;
173  char insc=' ';
174  for (Input_iterator it=first; it!=last;++it){
175  typename System::Residue& residue=chain.get_or_create_residue(resname,++res_index,insc);
176  residue.add_coarse_atom(*it,0);
177  }
178 }
