44  void even_layer(const FT& center_x,const FT& center_y,const FT& layer_z,const FT& unit_move,unsigned size,Output_iterator out){
45  const FT x_s=center_x - (size-1)*unit_move;
46  const FT y_s=center_y - (size-1)*unit_move;
47 
48  for (unsigned i=0;i<size;++i)
49  for (unsigned j=0;j<size;++j)

84  template<class Point_3,class Output_iterator>
85  void fcc_lattice(const Point_3& center, double radius, double min_edge_length,Output_iterator out){
86  double unit_move=sqrt(2*radius);
87  unsigned size = static_cast<unsigned>( ceil( min_edge_length/2./unit_move ) )+1;
88 
89  double z_s=center.z() - (size-1)*unit_move;
90  for (unsigned i=0;i<size;++i)
91  internal_fcc_lattice::even_layer<Point_3>(center.x(),center.y(),z_s+2*i*unit_move,unit_move,size,out);
92  for (unsigned i=0;i<size-1;++i)
93  internal_fcc_lattice::odd_layer<Point_3>(center.x(),center.y(),z_s+2*i*unit_move+unit_move,unit_move,size,out);
94  }
