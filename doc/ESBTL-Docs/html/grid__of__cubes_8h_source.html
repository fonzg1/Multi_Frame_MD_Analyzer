69  static void init_grid(const Iterator& begin,const Iterator& end,int& xlim,int& ylim,int& zlim,double& rmax,Point& lower_corner){
70  Point min,max;
71  boost::tie(min,max)=ESBTL::bounding_box<Point>(begin,end);
72 
73  //#warning should not be hardcoded
74  rmax = 3.;
75  xlim=static_cast<int>( ceil( ( max.x()-min.x() + 2*rmax )/rmax) );
76  ylim=static_cast<int>( ceil( ( max.y()-min.y() + 2*rmax )/rmax) );
77  zlim=static_cast<int>( ceil( ( max.z()-min.z() + 2*rmax )/rmax) );
78  lower_corner=min;
79  };

92  template<class Any_iterator>
93  static boost::tuple<int,int,int>
94  locate_cube(Any_iterator it,const int& xlim,const int& ylim,const int& zlim,const double& cubel,const Point& lower_corner){
95  int x=static_cast<int>( floor( ( it->x()-lower_corner.x() )/cubel ) );
96  int y=static_cast<int>( floor( ( it->y()-lower_corner.y() )/cubel ) );
97  int z=static_cast<int>( floor( ( it->z()-lower_corner.z() )/cubel ) );
98 
99  //handle limit cases: snapping point at the limit of the bounding box and possibly
100  //intersecting an element of the cube. This is not the optimal method as we are making more test than needed
101  //(we know that some cubes are free from intersection) : maybe allow this in the iterator: just modify the is_outside_grid
102  //criteria and allow iterator for *lim and -1
103  if (x==xlim) x=xlim-1;
104  if (y==ylim) y=ylim-1;
105  if (z==zlim) z=zlim-1;
106  if (x==-1) x=0;
107  if (y==-1) y=0;
108  if (z==-1) z=0;
109 
110  assert(!(x<0) && x < xlim);
111  assert(!(y<0) && y < ylim);
112  assert(!(z<0) && z < zlim);
113  return boost::make_tuple(x,y,z);
114  };

126  template<class Any_iterator>
127  static bool
128  is_outside_grid(Any_iterator it,const int& xlim,const int& ylim,const int& zlim,const double& cubel,const Point& lower_corner){
129  int x=static_cast<int>( floor( ( it->x()-lower_corner.x() )/cubel ) );
130  int y=static_cast<int>( floor( ( it->y()-lower_corner.y() )/cubel ) );
131  int z=static_cast<int>( floor( ( it->z()-lower_corner.z() )/cubel ) );
132  if (x<-1 || x > xlim) return true;
133  if (y<-1 || y > ylim) return true;
134  if (z<-1 || z > zlim) return true;
135  return false;
136  };

248  Cube_coordinates
249  get_next(){

251  Cube_coordinates ret=current;
252  if (current.get<0>() < grid_ptr->xlim-1)
253  ret.get<0>()++;
254  else{
255  if (current.get<1>() < grid_ptr->ylim-1){
256  ret.get<1>()++;
257  ret.get<0>()=0;
258  }
259  else{
260  if (current.get<2>() < grid_ptr->zlim-1){
261  ret.get<2>()++;
262  ret.get<0>()=0;
263  ret.get<1>()=0;
264  }
265  else{
266  return Cube_coordinates(-1,-1,-1);
267  }
268  }
269  }
270  if (grid_ptr->valid_tuple(ret)) return ret;
271  current=ret;
272  return get_next();
273  }

275  iterator(){};
276  iterator(const Cube_coordinates& C):current(C),grid_ptr(NULL){};
277  iterator(Grid_of_cubes* Grd):grid_ptr(Grd){current=make_tuple(-1,-1,-1);};
278  iterator(Grid_of_cubes* Grd,const Cube_coordinates& C):current(C),grid_ptr(Grd){};
279  iterator& operator++(){
280  current=get_next();

292  assert(it!=grid_ptr->cube_container.end());//DEBUG
293  return (*it).second;
294  }
295  bool operator==(const iterator& it){
296  return this->current==it.current;
297  }
298  bool operator!=(const iterator& it){
299  return this->current!=it.current;

303  class neighbor_iterator : public iterator
304  {
305  using iterator::current;
306  using iterator::grid_ptr;
307  private:
308  Cube_coordinates center;
309 
310  Cube_coordinates get_next(){
311  Cube_coordinates ret=current;
312  assert(current!=make_tuple(-1,-1,-1));//DEBUG
313  if ( (current.template get<0>() < center.get<0>() + 1) && (current.template get<0>() < grid_ptr->xlim - 1) )
314  ret.get<0>()++;
315  else{
316  if ( (current.template get<1>() < center.get<1>() + 1 ) && (current.template get<1>() < grid_ptr->ylim-1) ){
317  ret.get<1>()++;
318  ret.get<0>()=((center.get<0>() == 0)?(center.get<0>()):(center.get<0>()-1));
319  }
320  else{
321  if ( (current.template get<2>() < center.get<2>()+1) && (current.template get<2>() < grid_ptr->zlim-1) ){
322  ret.get<2>()++;
323  ret.get<0>()=((center.get<0>() == 0)?(center.get<0>()):(center.get<0>()-1));
324  ret.get<1>()=((center.get<1>() == 0)?(center.get<1>()):(center.get<1>()-1));
325  }
326  else{
327  return make_tuple(-1,-1,-1);
328  }
329  }
330  }
331  if (ret==center){
332  current=ret;
333  return get_next();
334  }
335  if (grid_ptr->valid_tuple(ret)) return ret;
336  current=ret;
337  return get_next();
338  }
339 
340  public:
341  neighbor_iterator(){}
342  neighbor_iterator(const Cube_coordinates& C):iterator(C){}
343  neighbor_iterator(const Cube_coordinates& C,Grid_of_cubes* Grd):iterator(Grd),center(C){
344  current=grid_ptr->get_first_neighbor(center);
345  if (!grid_ptr->valid_tuple(current))
346  current=this->get_next();
347  }
348 
349  Cube_coordinates& get_current(){
350  return current;
351  }
352 
353  neighbor_iterator& operator++(){
354  current=this->get_next();
355  return *this;
356  }
357  };
