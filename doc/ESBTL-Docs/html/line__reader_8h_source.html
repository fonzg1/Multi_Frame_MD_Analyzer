54  template <class Occupancy_handler,class Ifstream>
55  bool read_stream(Ifstream& input,Occupancy_handler occupancy,char default_altloc=char(' ')){
56  int nblines=0;
57 
58  while ( ! input.eof() ){
59  std::string line;
60  getline(input,line);
61  if (line.empty()) continue;
62 
63 
64  Line_format line_format(line);
65 
66  int system_index=line_selector.keep(line_format,line,occupancy);
67 
68  if (system_index!=DISCARD){
69  //TODO warning Put inside selector: we could think of a different policy
70  char altloc=line_format.get_alternate_location(line);
71  if (altloc!=' '){
72  if (default_altloc==' ')
73  default_altloc=altloc;
74  else
75  if (altloc!=default_altloc) continue;
76  }
77 
78  builder.interpret_line(line_format,line,system_index);
79  ++nblines;
80  }
81  }
82 
83  nblines+=occupancy.finalize(builder);
84  builder.create_systems(default_altloc);
85 
86  #ifndef NDEBUG
87  std::cout << "(ESBTL-DEBUG) Lines read " << nblines << std::endl;
88  #endif
89  return true;
90  }

114  template <Reading_mode mode,class Occupancy_handler>
115  bool read(const std::string& filename,Occupancy_handler occupancy,char default_altloc=' '){
116  //binary flags prevent from interpretation of some symbols \r\n for example
117  std::ifstream input( filename.c_str(),std::ios_base::binary );
118 
119  if (! input){
120  std::cerr << "Problem while trying to open file " << filename << ".\nPlease check that the file exists and that you have the right to read it." << std::endl;
121  return false;
122  }
123 
124  internal::Ifstream_compress<mode> stream;
125  return read_stream(stream.get(input),occupancy,default_altloc);
