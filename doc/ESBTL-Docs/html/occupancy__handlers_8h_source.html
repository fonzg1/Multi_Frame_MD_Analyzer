120  template <class Line_format,bool take_max>
121  class Min_or_max_occupancy_policy{
122  protected:
123  typedef std::map<double,std::pair<int,std::list<std::string> > > Info;
124  Info lines;
125  double min_or_max_occupancy;
126 
127  Min_or_max_occupancy_policy():min_or_max_occupancy(take_max?0.:1.){}
128 
129  void add_line(double occupancy_value,const std::string& line,int system_index){
130  typename Info::iterator it=lines.find(occupancy_value);
131  if (it==lines.end())
132  it=lines.insert(
133  std::make_pair(occupancy_value,
134  std::make_pair(system_index,std::list<std::string>()))
135  ).first;
136  it->second.second.push_back(line);
137  }
138 
139  public:
140  int add_or_postpone(const Line_format& line_format,const std::string& line,int system_index){
141  double occupancy_value=line_format.get_occupancy(line);
142  if (occupancy_value==1. || line_format.get_alternate_location(line)!=' ')
143  return system_index;
144 
145  bool do_take_it=take_max?
146  occupancy_value >= min_or_max_occupancy:
147  occupancy_value <= min_or_max_occupancy;
148 
149  if (do_take_it){
150  add_line(occupancy_value,line,system_index);
151  min_or_max_occupancy=occupancy_value;
152  }
153 
154  return DISCARD;
155  }
156 
157  template <class Builder>
158  int finalize(Builder& builder) const {
159  if (lines.empty())
160  return 0;
161  if (min_or_max_occupancy==0.5){
162  std::cerr << "Fatal error: The atom occupancy selection policy is ambiguous is that case (0.5)\n";
163  exit(EXIT_FAILURE);
164  }
165  int lines_added=0;
166  Info::const_iterator it=lines.find(min_or_max_occupancy);
167  for (std::list<std::string>::const_iterator cur=it->second.second.begin();
168  cur!=it->second.second.end(); ++cur)
169  {
170  ++lines_added;
171  builder.interpret_line(Line_format(*cur),*cur,it->second.first);
172  }
173 
174  return lines_added;
175  }
176  };
