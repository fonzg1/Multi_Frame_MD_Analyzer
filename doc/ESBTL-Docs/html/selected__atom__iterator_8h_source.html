71  //functor may be initialized with info.
76  Selected_atom_iterator(Base_iterator it,const Subset_functor& functor):atom_iterator(it),keep(functor)
77  {if (!atom_iterator.is_end() && !keep(*atom_iterator)) increment();}

80  {if (!atom_iterator.is_end() && !keep(*atom_iterator)) increment();}

82 private:

85  void increment(){
86  do{
87  ++atom_iterator;
88  }
89  while(!atom_iterator.is_end() && !keep(*atom_iterator));
90  }
91

98  }
