91  template <class System_iterator>
92  System_updater_from_xdrfile(System_iterator begin, System_iterator end,unsigned model_selected_,const std::string& xtc_fname,
93  unsigned max_atoms,unsigned read_from=1,
94  unsigned read_to=std::numeric_limits<unsigned>::max() ):first_frame_id_(read_from),last_frame_id_(read_to),nb_atoms_to_read_(max_atoms),current_frame_(0)
95  {
96  input_file_=xdrfile_open(xtc_fname.c_str(), "r");
97  if (input_file_ == NULL){
98  std::cerr << "Error while opening xtc file " << xtc_fname << std::endl;
99  exit (EXIT_FAILURE);
100  }
101 
102  for (System_iterator it_sys=begin;it_sys!=end;++it_sys){
103  typename System::Model& model=it_sys->get_model(model_selected_);
104  for (typename System::Model::Atoms_iterator it_atm=model.atoms_begin();it_atm!=model.atoms_end();++it_atm){
105  assert( selected_atoms_.find(it_atm->atom_serial_number())==selected_atoms_.end() );
106  selected_atoms_.insert(std::make_pair(it_atm->atom_serial_number(),&(*it_atm)));
107  }
108  }
109 
110  //check the file contains at least one frame
111  int magic=0;
112  int result=xdrfile_read_int(&magic,1,input_file_);
113  if (result==0){
114  std::cerr << "xtc file provided contains no frame: error while reading magic number" << std::endl;
115  exit (EXIT_FAILURE);
116  }
117  assert (magic==1995); //checks XDR magic number
118 
119  if (first_frame_id_!=1)
120  next_frame(true);
121 
122  assert(current_frame_+1==first_frame_id_);
123  }
124 
125  ~System_updater_from_xdrfile(){
126  //only when reading stops at a given frame id
127  if (input_file_!=NULL)
128  xdrfile_close(input_file_);
129  }

132  //return true if next_frame can be loaded
133  //return in addition the corresponding simulation step and the simulation time.
139  boost::tuple<bool,double,int> next_frame(bool is_init=false){
140  if (!has_more_frames())
141  return boost::make_tuple(false,-1,-1);
142 
143  int magic,result, natoms,step;
144  float time, prec, box[9];//, min[3], max[3];
145  float xyz[3*nb_atoms_to_read_];
146 
147  do
148  {
149  ++current_frame_;
150  // Read the compressed xtc file
151  xdrfile_read_int(&natoms,1,input_file_); // reads number of atoms in the frame
152  assert(natoms == static_cast<int>(nb_atoms_to_read_)); // number of atoms in the PDB record matches the frame
153  xdrfile_read_int(&step,1,input_file_); // reads the step number
154  xdrfile_read_float(&time,1,input_file_);
155  xdrfile_read_float(box,9,input_file_); // reads the box dimensions
156  result = xdrfile_decompress_coord_float(xyz,&natoms,&prec,input_file_); //reads the coordinates in xyz
157  assert(result != 0);
158 
159  result=xdrfile_read_int(&magic,1,input_file_);
160  if (result==0){
161  if (current_frame_ <= first_frame_id_ - 1){
162  std::cerr << "Starting frame id is greater than the number of frames in the provided file" << std::endl;
163  exit(EXIT_FAILURE);
164  }
165  xdrfile_close(input_file_);
166  input_file_=NULL;
167  }
168  assert (result==0 || magic==1995); //checks XDR magic number
169  }
170  while( current_frame_ < first_frame_id_ - 1 );
171 
172  //update the coordinates
173  if (!is_init)
174  for(typename Map_id_to_atom::iterator it_sel= selected_atoms_.begin(); it_sel!=selected_atoms_.end();++it_sel){
175  unsigned i=it_sel->first;
176  typename System::Atom::Point_3 new_center(
177  xyz[3*(i-1)]*10,
178  xyz[3*(i-1)+1]*10,
179  xyz[3*(i-1)+2]*10
180  );//*10 because gromacs is in nanometer
181 
182  static_cast<typename System::Atom::Point_3&>(*it_sel->second)=new_center;
183  }
184  return boost::make_tuple(true,time,step);
185  }
186 
188  bool has_more_frames(){
189  return (input_file_!=NULL && current_frame_ !=last_frame_id_);
190  }
191 
192  const unsigned& current_frame_id(){return current_frame_;}
193  };
