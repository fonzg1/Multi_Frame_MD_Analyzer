94  template <class Iterator>
95  double rms_no_align(Iterator begin1,Iterator end1,Iterator begin2,Iterator end2){
96  typedef boost::tuple<Iterator,Iterator> It_tuple;
97  typedef boost::zip_iterator<It_tuple> Zip_iterator;
98  double nb=0;
99  double sum_sqd=0;
100  for (Zip_iterator it=boost::make_zip_iterator(boost::make_tuple(begin1,begin1));it!=boost::make_zip_iterator(boost::make_tuple(end1,end2));++it){
101  sum_sqd+=squared_distance(boost::get<0>(*it),boost::get<1>(*it));
102  nb+=1;
103  }
104  return sqrt(sum_sqd/nb);
105  }

116  std::pair<Point_3,Point_3>
117  bounding_box(Iterator begin,Iterator end){

119  double xmax=std::numeric_limits<double>::min(),ymax=std::numeric_limits<double>::min(),zmax=std::numeric_limits<double>::min();
120  for (Iterator it=begin;it!=end;++it){
121  if (it->x() < xmin) xmin=it->x();
122  if (it->x() > xmax) xmax=it->x();
123  if (it->y() < ymin) ymin=it->y();
124  if (it->y() > ymax) ymax=it->y();
125  if (it->z() < zmin) zmin=it->z();
126  if (it->z() > zmax) zmax=it->z();
127  }
128  return std::make_pair(Point_3(xmin,ymin,zmin),Point_3(xmax,ymax,zmax));

142  bounding_cube(Iterator begin,Iterator end){
143  std::pair<Point_3,Point_3> bbox=bounding_box<Point_3>(begin,end);
